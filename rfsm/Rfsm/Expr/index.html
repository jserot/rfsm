mutable e_typ : Types.typ;
}

| EInt of int
| EFloat of float
| EChar of char
| EBool of bool
| EEnum of string
| EVar of string
| EBinop of string * t * t

| ECond of t * t * t

e1 ? e2 : e3

| EFapp of string * t list

f(arg1,...,argn)

| EArrExt of t list

e1,...,e2

| EArr of string * t

| EBit of string * t
