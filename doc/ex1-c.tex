\chapter*{Appendix C1 - Example of generated C code}  
\label{cha:ex1-c}

This is the code generated from program given in Listing~\ref{lst:rfsm-gensig} by the C backend.

\begin{lstlisting}[language=ctask,frame=single,numbers=none,basicstyle=\small]
task Gensig<int n>(
  in event h;
  in bool e;
 out bool s;
  )
{
  int<1:n> k;
  enum { E0,E1 } state = E0;
  s = false;
  while ( 1 ) {
    switch ( state ) {
    case E0:
      wait_ev(h);
      if ( e==true ) {
        k = 1;
        s = true;
        state = E1;
        }
      break;
    case E1:
      wait_ev(h);
      if ( k==n ) {
        s = false;
        state = E0;
        }
      else if ( k<n ) {
        k = k+1;
        }
      break;
    }
  }
};
\end{lstlisting}
